#include <stdio.h>
            int main(){
              float valor, converte;
              char unidade;
            
              scanf("%f %c", &valor, &unidade);
            
              if (unidade == 'f' || unidade == 'F'){
                converte = (valor - 32.0)/1.8;
                printf("%.2f (C)", converte);
              }
              if (unidade == 'c' || unidade == 'C'){
                converte = (valor*1.8) + 32;
                printf("%.2f (F)", converte);
              }
              
              return 0;
            }
